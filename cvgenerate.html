function generateCV() { // Collect all form data const formData = { photo: document.getElementById('photo-preview').src, fullName: document.getElementById('full-name').value, jobTitle: document.getElementById('job-title').value, summary: document.getElementById('summary').value, mobile: document.getElementById('mobile').value, email: document.getElementById('email').value, address: document.getElementById('address').value, experiences: [], education: [], skills: [], certificates: [], references: [] }; // Collect experiences document.querySelectorAll('.experience-item').forEach(item => { formData.experiences.push({ company: item.querySelector('input[name="company[]"]').value, position: item.querySelector('input[name="position[]"]').value, duration: item.querySelector('input[name="duration[]"]').value, responsibilities: item.querySelector('textarea[name="responsibilities[]"]').value.split('\n') }); }); // Collect education document.querySelectorAll('.education-item').forEach(item => { formData.education.push({ degree: item.querySelector('input[name="degree[]"]').value, institution: item.querySelector('input[name="institution[]"]').value, duration: item.querySelector('input[name="education-duration[]"]').value, grade: item.querySelector('input[name="grade[]"]').value }); }); // Collect skills document.querySelectorAll('.skill-item').forEach(item => { formData.skills.push(item.querySelector('input[name="skills[]"]').value.split(',')); }); // Collect certificates document.querySelectorAll('.certificate-item').forEach(item => { formData.certificates.push({ year: item.querySelector('input[name="cert-year[]"]').value, name: item.querySelector('input[name="cert-name[]"]').value }); }); // Collect references const refNames = document.querySelectorAll('input[name="ref-name[]"]'); const refContacts = document.querySelectorAll('input[name="ref-contact[]"]'); const refAddresses = document.querySelectorAll('input[name="ref-address[]"]'); for (let i = 0; i < refNames.length; i++) { formData.references.push({ name: refNames[i].value, contact: refContacts[i].value, address: refAddresses[i].value }); } // Store in sessionStorage to pass to the next page sessionStorage.setItem('cvData', JSON.stringify(formData)); // Redirect to preview page window.location.href = 'cvgenerate.html'; }
